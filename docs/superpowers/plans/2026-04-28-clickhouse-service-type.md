# ClickHouse Service Type Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add ClickHouse as a supported `unit-operator` service type with UnitSet lifecycle support and ClickHouse agent operations for logical backup, restore, and dynamic settings.

**Architecture:** Follow the existing per-engine agent pattern used by MySQL, PostgreSQL, MongoDB, Redis, and Milvus. Add a new `pkg/agent/app/clickhouse` gRPC app, expose it through `GrpcCall` for `type=clickhouse`, and leave `clickhouse-keeper` as lifecycle/config-sync only through the generic UnitSet/Unit path.

**Tech Stack:** Go, controller-runtime, Kubernetes CRDs via controller-gen, protobuf/gRPC via `make pb-gen`, ClickHouse `clickhouse-client`, existing unit-agent SLM and config sync services.

---

package clickhouse

import (
	"context"
	"fmt"
	"net/url"
	"os"
	"os/exec"
	"regexp"
	"strings"

	"github.com/upmio/unit-operator/pkg/agent/app"
	"github.com/upmio/unit-operator/pkg/agent/app/common"
	"github.com/upmio/unit-operator/pkg/agent/app/slm"
	"github.com/upmio/unit-operator/pkg/agent/pkg/util"
	"go.uber.org/zap"
	"google.golang.org/grpc"
)

const (
	clickHouseHostEnvKey   = "CLICKHOUSE_HOST"
	clickHousePortEnvKey   = "CLICKHOUSE_PORT"
	clickHouseSecureEnvKey = "CLICKHOUSE_SECURE"
)

var (
	svr = &service{}

	variableKeyRegexp = regexp.MustCompile(`^[A-Za-z_][A-Za-z0-9_.]*$`)
)

type commandRunner interface {
	ExecuteCommand(cmd *exec.Cmd, logPrefix string) error
}

type commonCommandRunner struct {
	logger *zap.SugaredLogger
}

func (r commonCommandRunner) ExecuteCommand(cmd *exec.Cmd, logPrefix string) error {
	return common.NewCommandExecutor(r.logger).ExecuteCommand(cmd, logPrefix)
}

type clickHouseConnection struct {
	host   string
	port   string
	secure bool
}

type service struct {
	chOps ClickHouseOperationServer
	UnimplementedClickHouseOperationServer
	logger *zap.SugaredLogger
	slm    slm.ServiceLifecycleServer
	runner commandRunner
}

func (s *service) Config() error {
	s.chOps = app.GetGrpcApp(appName).(ClickHouseOperationServer)
	s.logger = zap.L().Named(appName).Sugar()
	s.slm = app.GetGrpcApp("slm").(slm.ServiceLifecycleServer)
	s.runner = commonCommandRunner{logger: s.logger}
	return nil
}

func (s *service) Name() string {
	return appName
}

func (s *service) Registry(server *grpc.Server) {
	RegisterClickHouseOperationServer(server, svr)
}

func (s *service) LogicalBackup(ctx context.Context, req *LogicalBackupRequest) (*common.Empty, error) {
	if err := s.checkProcessStarted(ctx); err != nil {
		return nil, err
	}
	if err := validateUsername(req.GetUsername()); err != nil {
		return nil, err
	}
	sql, err := buildBackupSQL(req.GetBackupFile(), req.GetObjectStorage())
	if err != nil {
		return nil, err
	}
	password, err := util.DecryptPlainTextPassword(req.GetUsername())
	if err != nil {
		return nil, err
	}
	if err := s.runClickHouseQuery(ctx, readClickHouseConnection(), req.GetUsername(), password, sql); err != nil {
		return nil, err
	}
	return nil, nil
}

func (s *service) Restore(ctx context.Context, req *RestoreRequest) (*common.Empty, error) {
	if err := s.checkProcessStarted(ctx); err != nil {
		return nil, err
	}
	if err := validateUsername(req.GetUsername()); err != nil {
		return nil, err
	}
	sql, err := buildRestoreSQL(req.GetBackupFile(), req.GetObjectStorage())
	if err != nil {
		return nil, err
	}
	password, err := util.DecryptPlainTextPassword(req.GetUsername())
	if err != nil {
		return nil, err
	}
	if err := s.runClickHouseQuery(ctx, readClickHouseConnection(), req.GetUsername(), password, sql); err != nil {
		return nil, err
	}
	return nil, nil
}

func (s *service) SetVariable(ctx context.Context, req *SetVariableRequest) (*common.Empty, error) {
	if err := s.checkProcessStarted(ctx); err != nil {
		return nil, err
	}
	if err := validateUsername(req.GetUsername()); err != nil {
		return nil, err
	}
	sql, err := buildSetVariableSQL(req.GetUsername(), req.GetKey(), req.GetValue())
	if err != nil {
		return nil, err
	}
	password, err := util.DecryptPlainTextPassword(req.GetUsername())
	if err != nil {
		return nil, err
	}
	if err := s.runClickHouseQuery(ctx, readClickHouseConnection(), req.GetUsername(), password, sql); err != nil {
		return nil, err
	}
	return nil, nil
}

func (s *service) checkProcessStarted(ctx context.Context) error {
	if s.slm == nil {
		return nil
	}
	if _, err := s.slm.CheckProcessStarted(ctx, nil); err != nil {
		return err
	}
	return nil
}

func (s *service) runClickHouseQuery(ctx context.Context, conn clickHouseConnection, username, password, query string) error {
	args := []string{
		"--host", conn.host,
		"--port", conn.port,
		"--user", username,
		"--password", password,
	}
	if conn.secure {
		args = append(args, "--secure")
	}
	args = append(args, "--query", query)

	cmd := exec.CommandContext(ctx, "clickhouse-client", args...)
	return s.runner.ExecuteCommand(cmd, appName)
}

func readClickHouseConnection() clickHouseConnection {
	host := strings.TrimSpace(os.Getenv(clickHouseHostEnvKey))
	if host == "" {
		host = "127.0.0.1"
	}
	port := strings.TrimSpace(os.Getenv(clickHousePortEnvKey))
	if port == "" {
		port = "9000"
	}
	secure := strings.EqualFold(strings.TrimSpace(os.Getenv(clickHouseSecureEnvKey)), "true")
	return clickHouseConnection{host: host, port: port, secure: secure}
}

func buildBackupSQL(backupFile string, storage *common.ObjectStorage) (string, error) {
	s3URL, err := buildS3URL(storage, backupFile)
	if err != nil {
		return "", err
	}
	if err := validateObjectStorageCredentials(storage); err != nil {
		return "", err
	}
	return fmt.Sprintf("BACKUP ALL TO S3(%s, %s, %s)",
		quoteSQLString(s3URL),
		quoteSQLString(storage.GetAccessKey()),
		quoteSQLString(storage.GetSecretKey()),
	), nil
}

func buildRestoreSQL(backupFile string, storage *common.ObjectStorage) (string, error) {
	s3URL, err := buildS3URL(storage, backupFile)
	if err != nil {
		return "", err
	}
	if err := validateObjectStorageCredentials(storage); err != nil {
		return "", err
	}
	return fmt.Sprintf("RESTORE ALL FROM S3(%s, %s, %s)",
		quoteSQLString(s3URL),
		quoteSQLString(storage.GetAccessKey()),
		quoteSQLString(storage.GetSecretKey()),
	), nil
}

func buildSetVariableSQL(username, key, value string) (string, error) {
	if err := validateUsername(username); err != nil {
		return "", err
	}
	if err := validateVariableKey(key); err != nil {
		return "", err
	}
	if strings.TrimSpace(value) == "" {
		return "", fmt.Errorf("value is required")
	}
	return fmt.Sprintf("ALTER USER %s SETTINGS %s = %s", username, key, quoteSQLString(value)), nil
}

func buildS3URL(storage *common.ObjectStorage, backupFile string) (string, error) {
	if storage == nil {
		return "", fmt.Errorf("object_storage is required")
	}
	endpoint := strings.TrimRight(strings.TrimSpace(storage.GetEndpoint()), "/")
	if endpoint == "" {
		return "", fmt.Errorf("object_storage.endpoint is required")
	}
	bucket := strings.Trim(strings.TrimSpace(storage.GetBucket()), "/")
	if bucket == "" {
		return "", fmt.Errorf("object_storage.bucket is required")
	}
	object := strings.Trim(strings.TrimSpace(backupFile), "/")
	if object == "" {
		return "", fmt.Errorf("backup_file is required")
	}
	base, err := url.Parse(endpoint)
	if err != nil {
		return "", fmt.Errorf("parse object_storage.endpoint: %w", err)
	}
	segments := []string{strings.Trim(base.Path, "/"), bucket, object}
	joined := []string{}
	for _, segment := range segments {
		if segment != "" {
			joined = append(joined, segment)
		}
	}
	base.Path = "/" + strings.Join(joined, "/")
	return base.String(), nil
}

func validateObjectStorageCredentials(storage *common.ObjectStorage) error {
	if strings.TrimSpace(storage.GetAccessKey()) == "" {
		return fmt.Errorf("object_storage.access_key is required")
	}
	if strings.TrimSpace(storage.GetSecretKey()) == "" {
		return fmt.Errorf("object_storage.secret_key is required")
	}
	return nil
}

func validateUsername(username string) error {
	if !variableKeyRegexp.MatchString(username) {
		return fmt.Errorf("username contains unsupported characters")
	}
	return nil
}

func validateVariableKey(key string) error {
	if !variableKeyRegexp.MatchString(key) {
		return fmt.Errorf("key contains unsupported characters")
	}
	return nil
}

func quoteSQLString(value string) string {
	escaped := strings.ReplaceAll(value, `\`, `\\`)
	escaped = strings.ReplaceAll(escaped, `'`, `\'`)
	return "'" + escaped + "'"
}

func RegistryGrpcApp() {
	app.RegistryGrpcApp(svr)
}
